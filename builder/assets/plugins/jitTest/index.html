插件描述：JavaScript InfoVis工具包，用于创建网页交互式数据可视化工具。如果你想要直观的显示组织机构详情，那么可以使用这个插件。

下载插件   官方demos   官方文档

应用方法：

引入资源

<script src="js/jquery-3.2.1.min.js" type="text/javascript" charset="utf-8"></script>
<script language="javascript" type="text/javascript" src="./js/jit-yc.js"></script>
<script language="javascript" type="text/javascript" src="./js/index.js"></script>

如果你想提高效率，可以引入jquery加速开发-我这里应用了它

html代码

<div id="infovis"></div>

js代码-初始化

$(function(){
 init({
  nodeHeight: 20,
  nodeWidth: 50,
  nodeBgColor: 'blue'
 });
})
init({})目前只支持这三个参数：nodeWidth:节点宽度,nodeHeight:节点高度,nodeBgColor:节点背景色。后续你可以手动在index.js内部去更改或者添加自己应用时具体需要更改的对应参数即可

index.js内部init代码-说明

// 声明jit.ST组织树
var st= new $jit.ST({});
// 加载json数据
st.loadJSON(json);
// 计算布局
st.compute();
// 整体渲染动画
st.geom.translate(new $jit.Complex(-100, 0), "current");
// 触发根节点事件
st.onClick(st.root);

index.js内部new $jit.ST({})代码-说明

//渲染容器-对应html内部id
'injectInto': 'infovis',
//展示出拓扑图的动画时间 
duration: 300,
// 运动方式
transition: $jit.Trans.Quart.easeInOut,
//设置横向节点距离 
levelDistance: 30,
//设置纵向节点距离
siblingOffset: 15,
//点击节点后下级几点展示几层
levelsToShow: 1,
//节点类型样式设置，节点类型支持扩展 
Node: {
 height: nodeHeight, //节点高度
 width: nodeWidth, //节点宽度
// 节点背景类型circle, ellipse
 type: 'ellipse',
 color: nodeBgColor, //节点背景颜色
 align:"center", //居中
 overridable: true
},
Edge: {//线条类型(bezier,line,arrow,quadratic:begin,quadratic:end)样式设置
 type: 'bezier',
 lineWidth: 2, //线条宽度
 color:'#23A4FF', //线条颜色
 overridable: true
},
//是否可以移动  
Navigation: {
 enable: true,
 panning: true // true/false 是否可拖拽
},
request: function(nodeId, level, onComplete) {// 异步加载数据
},
onBeforeCompute: function(node){
// 正在创建中...
 Log.write("loading " + node.name);
},
onAfterCompute: function(){
// 创建已完成
 Log.write("done");
},
onCreateLabel: function(label, node){
 label.id = node.id;
 label.innerHTML = node.name;
 label.onclick = function(){
  // 给节点添加点击事件
  st.onClick(node.id);
  if (fn){
   fn(node.id, node.name);
  }
};
// jit支持自己的一些css样式写法
var style = label.style;
style.width = nodeWidth + 'px';
style.height = nodeHeight + 'px'; 
style.cursor = 'pointer';
style.color = '#fff';
//style.backgroundColor = '#1a1a1a';
style.fontSize = '0.8em';
style.textAlign= 'center';
// 也可以应用jquery的写法label既是选中元素
$(label).css('line-height', '20px')
},
onBeforePlotNode: function(node){
 if (node.selected) {
// 选中时，节点背景色
  node.data.$color = "#f60";
 }
 else {
  // 清除其他背景色
  delete node.data.$color;
 }
},
onBeforePlotLine: function(adj){
 if (adj.nodeFrom.selected && adj.nodeTo.selected) {
 // 选中时连线的颜色
 }
 else {
  delete adj.data.$color;
  delete adj.data.$lineWidth;
 }
}
